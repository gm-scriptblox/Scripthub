const params = new URLSearchParams(window.location.search); const pasteID = params.get("id"); if (!pasteID) { document.getElementById("content").textContent = "No paste ID provided."; } else { fetch("https://api.allorigins.win/raw?url=" + encodeURIComponent("https://paste.rs/" + pasteID)) .then(r => r.text()) .then(t => document.getElementById("content").textContent = t) .catch(e => document.getElementById("content").textContent = "Error fetching paste."); }
